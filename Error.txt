                async doQuery() {
                    if (!this.qry.lotId.trim()) {
                        VMessageBox.alert('Pls input Query Criterial !!<br>請輸入查詢條件 !!');
                        return;
                    }

                    document.body.style.cursor = 'wait';
                    this.setStatus('QUERYING...', 'info');

                    try {
                        let url = "";
                        let params = {};

                        // 判斷查詢情境與對應的 API
                        if (this.qry.queryMode === 'Current' || this.qry.queryMode === 'History') {
                            this.currentView = 'LotInfo';
                            this.dataLotInfo = [];
                            this.$refs.resultLotInfo.setData([]);

                            url = "/api/query/ws-merge-query/query";
                            params = {
                                LotId: this.qry.lotId,
                                IsHistoryTab: this.qry.queryMode === 'History'
                            };
                        } else {
                            this.currentView = 'Sorter';
                            this.dataSorter = [];
                            this.$refs.resultSorter.setData([]);

                            url = "/api/query/ws-merge-query/sorter-control";
                            params = {
                                LotId: this.qry.lotId,
                                Action: this.qry.queryMode // 'Merge' 或 'Split'
                            };
                        }

                        // 呼叫我們先前開發的 C# API
                        const response = await mesfetch(url, params);

                        // 依據視窗塞入資料
                        if (this.currentView === 'LotInfo') {
                            this.dataLotInfo = response;
                            this.$refs.resultLotInfo.setData(response);
                        } else {
                            this.dataSorter = response;
                            this.$refs.resultSorter.setData(response);
                        }

                        this.setStatus('QUERY SUCCESS', 'success');

                    } catch (error) {
                        console.error(error);
                        VMessageBox.alert(error.Message || '查詢發生預期外的錯誤', 'Error', {
                            icon: 'mdi-alert-circle', color: '#ed4245', iconBg: 'rgba(237,66,69,.1)'
                        }).catch(() => { });
                        this.setStatus('QUERY FAILED', 'error');
                    } finally {
                        document.body.style.cursor = 'default';
                    }
                },
